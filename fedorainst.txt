=============================== install ====================================
# speed up dnf
sudo dnf install nano

sudo nano /etc/dnf/dnf.conf
      [main]
      gpgcheck=1
	  installonly_limit=3
	  clean_requirements_on_remove=True
	  best=False
	  skip_if_unavailable=True
	  # added for speed:
	  fatsestmirror=True
	  max_parallel_downloads=10
	  defaultyes=True
	  keepcache=True

sudo dnf update

# install display manager
sudo dnf install sddm
